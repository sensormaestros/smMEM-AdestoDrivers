331 /***************************************************************************//**
332  * @brief
333  * Serial RX IRQ Handler
334  *
335  ******************************************************************************/
336 void SERIAL_RX_IRQHandler(void)
337 {
338  if (SPORT->IF & IF_RXDATAV)
339  {
340  uint8_t data = SPORT->RXDATA;
341  buf_write_byte(serial_rx_buf, data);
342  if (buf_full(serial_rx_buf))
343  SPORT->IEN &= ~ IEN_RXDATAV; // disable rx interrupt
344  }
345 }
346 
347 /***************************************************************************//**
348  * @brief
349  * Serial TX IRQ Handler
350  *
351  ******************************************************************************/
352 void SERIAL_TX_IRQHandler(void)
353 {
354  if (SPORT->IF & IF_TXBL)
355  {
356  uint8_t data;
357  if (buf_read_byte(serial_tx_buf, & data))
358  SPORT->TXDATA = data;
359  else
360  SPORT->IEN &= ~ IEN_TXBL; // disable tx interrupt
361  }
362 }
