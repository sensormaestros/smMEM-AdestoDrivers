55 #define RTC_DIVIDER ( 32 )

57 #define RTC_CLOCK ( 32768U )

98  ******************************************************************************/
99 void oneshot_start_s(uint32_t s)
100 {
101  uint32_t ticks = MSEC_TO_TICKS(1000 * s);
102 
103  oneshot_done_flag = false;
104 
105  if (ECODE_EMDRV_RTCDRV_OK != RTCDRV_StartTimer(oneshot_timer,
106  rtcdrvTimerTypeOneshot,
107  ticks,
108  oneshot_callback,
109  0))
110  fatal("can't initialize oneshot timer");
111 }
