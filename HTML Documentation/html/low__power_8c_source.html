77 /***************************************************************************//**
78  * @brief
79  * Decides whether to put EFM into EM0, EM1, or EM2 based on Activity
80  *
81  ******************************************************************************/
82 void enter_low_power_state(void)
83 {
84  if (force_em0)
85  __ASM volatile ("wfi");
86  else if (spi_active())
87  {
88  wait_em1_count++;
89  GPIO_PinOutSet(EM1_DEBUG_PORT, EM1_DEBUG_PIN);
90  EMU_EnterEM1();
91  GPIO_PinOutClear(EM1_DEBUG_PORT, EM1_DEBUG_PIN);
92  }
93  else
94  {
95  wait_em2_count++;
96  GPIO_PinOutSet(EM2_DEBUG_PORT, EM2_DEBUG_PIN);
97  EMU_EnterEM2(true);
98  GPIO_PinOutClear(EM2_DEBUG_PORT, EM2_DEBUG_PIN);
99  }
100 }
