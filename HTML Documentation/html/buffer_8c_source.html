50 typedef struct buf
51 {
52  size_t size;
53  volatile size_t count;
54  size_t write_idx;
55  size_t read_idx;
56  uint8_t data[0];
57 } buf_t;

59 /****************************************************************************//**
60  * @brief Buffer Initialization
61  *
62  * @param[in] *raw_buf
63  * Pointer to buffer
64  * @param[in] size
65  * Size of Buffer
66  * @return buf
67  *
68  *****************************************************************************/
69 buf_t *init_buf(uint8_t *raw_buf, size_t size)
70 {
71  buf_t *buf = (buf_t *) raw_buf;
72  memset(buf, 0, size);
73  buf->size = size - sizeof(buf_t);
74  buf->count = 0;
75  buf->write_idx = 0;
76  buf->read_idx = 0;
77  return buf;
78 }

106 /****************************************************************************//**
107  * @brief Buffer Write Byte
108  *
109  * @param[in] *buf
110  * Pointer to buffer
111  * @param[in] data
112  * Byte/data to write
113  * @return true if successful
114  *
115  *****************************************************************************/
116 bool buf_write_byte(buf_t *buf, const uint8_t data)
117 {
118  if (buf->count >= buf->size)
119  return false; // no room
120  buf->data[buf->write_idx++] = data;
121  if (buf->write_idx >= buf->size)
122  buf->write_idx = 0;
123  INT_Disable();
124  buf->count++;
125  INT_Enable();
126  return true;
127 }

129 /****************************************************************************//**
130  * @brief Buffer Read Byte
131  *
132  * @param[in] *buf
133  * Pointer to buffer
134  * @param[in] *data
135  * Pointer to data variable to write to
136  * @return true if successful
137  *
138  *****************************************************************************/
139 bool buf_read_byte(buf_t *buf, uint8_t *data)
140 {
141  if (buf->count == 0)
142  return false; // no data
143  *data = buf->data[buf->read_idx++];
144  if (buf->read_idx >= buf->size)
145  buf->read_idx = 0;
146  INT_Disable();
147  buf->count--;
148  INT_Enable();
149  return true;
150 }
