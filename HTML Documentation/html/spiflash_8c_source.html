257 /***************************************************************************//**

263  * @param[in] addr
264  * Address to check Power of 2 alignment
265  * @param[in] alignment_size

272 {

304  * @note

309  * Command to send
310  * @param[in] rx_len

324 {

338  NULL);

394  * @brief
395  * SPI Flash Command with Address
396  * @note
397  * spiflash_multiple_command_with_adddress() can be used to issue commands
398  * of any length. The reading of the response will begin after the last command
399  * byte is written. If no response is to be read, pass rx_len == 0.
400  * @param[in] cmd
401  * Command to send
402  * @param[in] addr
403  * Address for command to use
404  * @param[in] dumm_bytes
405  * Dummy bytes
406  * @param[in] tx_len
407  * Length of command to send in bytes
408  * @param[in] *tx_buf
409  * Pointer to command buffer to send
410  * @param[in] rx_len
411  * Expected response length, set to 0 if no response is expected.
412  * @param[in] *rx_buf
413  * Pointer to Rx buffer where response is placed.
414  * @param[in] *completion
415  * Completion function for Completion State_Machine
416  * @param[in] *completion_ref
417  * Argument to be passed to completion callback function
418  ******************************************************************************/
419 void spiflash_command_with_address(uint8_t cmd,
420  uint32_t addr,
421  unsigned int dummy_bytes,
422  size_t tx_len,
423  uint8_t *tx_buf,
424  size_t rx_len,
425  uint8_t *rx_buf,
426  spiflash_completion_fn_t *completion,
427  void *completion_ref)
428 {
429  int i = 0;
430 
431  spiflash_scratch_buf[i++] = cmd;
432  if (spiflash_info->address_bytes >= 3)
433  spiflash_scratch_buf[i++] = addr >> 16;
434  if (spiflash_info->address_bytes >= 2)
435  spiflash_scratch_buf[i++] = (addr >> 8) & 0xff;
436  if (spiflash_info->address_bytes >= 1)
437  spiflash_scratch_buf[i++] = addr & 0xff;
438  memset(&spiflash_scratch_buf[i], 0, dummy_bytes);
439  i += dummy_bytes;
440 
441  spiflash_completion = completion;
442  spiflash_completion_ref = completion_ref;
443 
444  spiflash_busy = true;
445 
446  spi_xfer(i, spiflash_scratch_buf, // tx1
447  tx_len, tx_buf, // tx2
448  true, // half duplex
449  rx_len, rx_buf,
450  false, // hold cs active
451  spiflash_spi_simple_completion,
452  NULL);
453 
454  if (! completion)
455  while (spiflash_busy)
456  enter_low_power_state();
457 }

549 /***************************************************************************//**
550  * @brief
551  * SPI Erase Completion State 2
552  * @note
553  * Issues ERASE Commands and checks what type of Erase is available
554  * on Device, checks address alignment
555  * @param[in] *ref
556  * Completion Function Pointer
557  ******************************************************************************/
558 static void spiflash_erase_completion2(void *ref)
559 {
560  int i;
561 
562  if (erase_info_fixed)
563  erase_info = erase_info_fixed;
564  else
565  {
566  for (i = spiflash_info->erase_info_count - 1; i >= 0; i--)
567  {
568  erase_info = & spiflash_info->erase_info [i];
569  if ((erase_len >= erase_info->size) &&
570  addr_aligned(erase_addr, erase_info->size))
571  break;
572  }
573  if (i < 0)
574  {
575  // no suitable erase command found!
576  // should never happen
577  erase_info = NULL;
578  while (true)
579  ;
580  }
581  }
582  if (erase_info->addr_needed)
583  {
584  spiflash_command_with_address(erase_info->cmd,
585  erase_addr,
586  0, // dummy bytes
587  0, NULL, // tx
588  0, NULL, // rx
589  spiflash_erase_completion3,
590  NULL);
591  }
592  else if (spiflash_info->dataflash)
593  {
594  // chip erase for dataflash doesn't need an address, but is a multibyte command
595  spiflash_multiple_byte_command(sizeof(dataflash_cmd_chip_erase), dataflash_cmd_chip_erase,
596  0, NULL, // rx
597  false, // hold_cs_active
598  spiflash_erase_completion3,
599  NULL);
600  }
601  else
602  {
603  // chip erase doesn't need an address
604  spiflash_single_byte_command(erase_info->cmd,
605  0, NULL, // rx

691 /***************************************************************************//**
692  * @brief
693  * SPI Flash Erase
694  * @note
695  * Top Level SPI Erase function, steps through Erase completion states.
696  * Determines if Active SO is to be used or not. Checks address alignment.
697  * Enters Low Power State during Erase.
698  * The erase commands will be ignored by the flash chip unless it has
699  * been previously write-enabled. Will return false, and never call
700  * completion function, if address is not erase page aligned or len is not
701  * a multiple of erase page size.
702  * @param[in] addr
703  * Address to read from
704  * @param[in] len
705  * How many bytes to read
706  * @param[in] cmd_size
707  * Size of Command in Bytes
708  * @param[in] use_so_irq
709  * Determines whether or not Active SO is used or not.
710  * @param[in] *completion
711  * Completion Function pointer.
712  * @param[in] *completion_ref
713  * Argument to be passed to completion callback function
714  * @return true
715  * If successful
716  ******************************************************************************/
717 bool spiflash_erase(uint32_t addr,
718  size_t len,
719  uint32_t cmd_size, // bytes per erase command, 0 for auto
720  bool use_so_irq,
721  spiflash_completion_fn_t *completion,
722  void *completion_ref)
723 {
724  int i;
725  bool found = false;
726 
727  spiflash_use_so_irq = use_so_irq && spiflash_info->has_so_irq;
728 
729  if (cmd_size)
730  {
731  // The desired erase size per command was specified.
732  // Find the matching erase command, and ensure the address is
733  // suitably aligned and the erase size is a multiple of the
734  // command size.
735  if (len % cmd_size)
736  return false;
737  for (i = spiflash_info->erase_info_count - 1; i >= 0; i--)
738  {
739  erase_info_fixed = & spiflash_info->erase_info [i];
740  if (cmd_size == erase_info_fixed->size)
741  {
742  found = true;
743  break;
744  }
745  }
746  if (! found)
747  {
748  erase_info_fixed = NULL;
749  return false;
750  }
751  }
752  else
753  {
754  // No erase size per command was specified.
755  // Ensure that there is a suitable choice.
756  for (i = spiflash_info->erase_info_count - 1; i >= 0; i--)
757  {
758  erase_info = & spiflash_info->erase_info [i];
759  if (len % erase_info->size)
760  continue;
761  if (! addr_aligned(addr, erase_info->size))
762  continue;
763  found = true;
764  break;
765  }
766  erase_info = NULL; // will be chosen in spiflash_erase_completion2;
767  if (! found)
768  return false;
769  }
770 
771  erase_addr = addr;
772  erase_len = len;
773  spiflash_completion = completion;
774  spiflash_completion_ref = completion_ref;
775  spiflash_erase_busy = true;
776 
777  if (spiflash_info->dataflash)
778  spiflash_multiple_byte_command(sizeof(dataflash_cmd_disable_sector_protection), dataflash_cmd_disable_sector_protection,
779  0, NULL, // rx
780  false, // hold_cs_active
781  spiflash_erase_completion1,
782  NULL);
783 
784  else
785  spiflash_erase_completion1(NULL);
786 
787  // if called synchronously, wait for entire write sequence to complete
788  if (! completion)
789  {
790  while (spiflash_erase_busy)
791  {
792  enter_low_power_state();
793  }
794  }
795  return true;
796 }
